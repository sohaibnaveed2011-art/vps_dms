services:
  admin_db:
    image: mysql:8.0
    container_name: ${DB_CONTAINER:-epos-db-rest2}
    restart: always
    environment:
      MYSQL_DATABASE: ${DB_NAME:-dms}
      MYSQL_USER: ${DB_USER:-eposuser}
      MYSQL_PASSWORD: ${DB_PASS:-epospass}
      MYSQL_ROOT_PASSWORD: ${ROOT_PASS:-rootpass}
    volumes:
      - dbdata:/var/lib/mysql
    networks:
      - epos-net

  admin:
    build: .
    container_name: dms_admin_app
    restart: always
    tty: true
    working_dir: /app
    volumes:
      - ./:/app
    ports:
      - "8000:8000"
    develop:
      watch:
        - action: sync
          path: ./
          target: /var/www/html
          ignore:
            - node_modules
            - vendor
        - action: rebuild
          path: package.json
    environment:
      DB_HOST: ${DB_CONTAINER:-epos-db-rest2}
      DB_PORT: 3306
      DB_DATABASE: ${DB_NAME:-dms}
      DB_USERNAME: ${DB_USER:-eposuser}
      DB_PASSWORD: ${DB_PASS:-epospass}
      APP_ENV: production
      APP_DEBUG: "false"
    depends_on:
      - admin_db
    networks:
      - epos-net

  admin_queue:
    build: .
    container_name: dms_admin_queue
    restart: always
    command: php artisan queue:work --tries=3 --timeout=90
    volumes:
      - ./:/app
    environment:
      DB_HOST: ${DB_CONTAINER:-epos-db-rest2}
      DB_PORT: 3306
      DB_DATABASE: ${DB_NAME:-dms}
      DB_USERNAME: ${DB_USER:-eposuser}
      DB_PASSWORD: ${DB_PASS:-epospass}
    depends_on:
      - admin_db
    networks:
      - epos-net

networks:
  epos-net:
    external: true

volumes:
  dbdata:
    driver: local
